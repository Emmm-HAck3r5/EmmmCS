\subsection{接口}
\begin{tabular}{|c|c|c|c|}
    \hline
    类型    &   位宽    &   名称    &   说明\\\hline
    input   &   1   &   clk &   时钟\\\hline

    input   &   CPU\_GREGIDX\_WIDTH &   rs2\_idx    &   rs2下标\\\hline

\subsection{说明}
